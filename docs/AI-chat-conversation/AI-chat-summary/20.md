```typescript
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Validate input
        const sanitizedQuery = this.sanitizeQuery(query);
        if (!sanitizedQuery) {
            throw new ApiError(400, 'Search query cannot be empty', false);
        }

        // Check cache
        const cached = this.getFromCache(sanitizedQuery);
        if (cached) {
            return cached;
        }

        try {
            const response = await apiClient.searchDatasets(sanitizedQuery, limit);

            // Validate response structure
            if (!response || !Array.isArray(response.results)) {
                throw new ApiError(500, 'Invalid response structure', false);
            }

            // Cache result
            this.setCache(sanitizedQuery, response);
            return response;
        }
    }

    private sanitizeQuery(query: string): string {
        return query
            .trim()
            .replace(/\s+/g, ' ')
            .slice(0, 1000);
```

```typescript
export class SearchQueryValidator {
    static readonly MIN_LENGTH = 1;
    static readonly MAX_LENGTH = 1000;

    static validate(query: string): void {
        const trimmed = query.trim();

        if (trimmed.length === 0) {
            throw new ValidationError('query', 'Search query cannot be empty');
        }

        if (trimmed.length < this.MIN_LENGTH) {
            throw new ValidationError('query', `Query must be at least ${this.MIN_LENGTH} character`);
        }

        if (trimmed.length > this.MAX_LENGTH) {
            throw new ValidationError('query', `Query cannot exceed ${this.MAX_LENGTH} characters`);
        }
    }

    static sanitize(query: string): string {
        return query
            .trim()
            .replace(/\s+/g, ' ')
            .slice(0, this.MAX_LENGTH);
    }
}

export class ChatMessageValidator {
    static readonly MIN_LENGTH = 1;

        }
    }
```

## 2. API Client Validation (Contract Layer)

### 2.1 Why Validate at API Level?

The API client is the **boundary between frontend and backend**. Invalid responses indicate either:
- Backend bug
- Network corruption
- Attacker interference

```typescript
// BAD: Trust response structure
async function search(query: string): Promise<SearchResponse> {
    const response = await fetch('/api/search', { body: JSON.stringify({ query }) });
    return response.json();  // Could be anything!
}

// GOOD: Validate response structure
async function search(query: string): Promise<SearchResponse> {
    const response = await fetch('/api/search', { body: JSON.stringify({ query }) });
    const data = response.json();
    
    // Validate against expected schema
    if (!data.results || !Array.isArray(data.results)) {
        throw new Error('Invalid response structure');
    }
    
    return data as SearchResponse;
}
```

### 2.2 Runtime Validation Benefits

TypeScript types exist only at compile time. At runtime, data could be anything:

```typescript
// TypeScript can't catch this at runtime:
const response = await fetch('/api/search');
const data: SearchResponse = await response.json();  // Might not be SearchResponse!

// Better: Runtime validation
function isSearchResponse(data: unknown): data is SearchResponse {
    if (!data || typeof data !== 'object') return false;
    const obj = data as Record<string, unknown>;
    return (
        Array.isArray(obj.results) &&
        obj.results.every(r => typeof (r as any).id === 'string')
    );
}

const data = await response.json();
if (!isSearchResponse(data)) {
    throw new Error('Response does not match SearchResponse schema');
}
```

## 3. Service Layer Validation

### 3.1 Business Logic Validation

Service layer enforces business rules:

```typescript
class SearchApiService {
    async search(query: string, limit: number): Promise<SearchResponse> {
        // Business rule: query must not be empty
        if (!query.trim()) {
            throw new ApiError(400, 'Query cannot be empty', false);
        }

        // Business rule: limit bounds
        if (limit < 1 || limit > 100) {
            throw new ApiError(400, 'Limit must be 1-100', false);
        }

        // Business rule: prevent spam/DOS with query length
        if (query.length > 1000) {
            throw new ApiError(400, 'Query too long', false);
        }

        // If all valid, proceed
        return apiClient.searchDatasets(query, limit);
    }
}
```

### 3.2 Your Sanitization

Your service sanitizes input before sending:

```typescript
private sanitizeQuery(query: string): string {
    return query
        .trim()                    // Remove leading/trailing spaces
        .replace(/\s+/g, ' ')      // Collapse multiple spaces to one
        .slice(0, 1000);           // Enforce max length
}
```

**Why sanitize before API call?**
- ✅ Reduces data sent over network
- ✅ Prevents server from dealing with malformed data
- ✅ Improves caching (normalized queries)

## 4. Component/UI Validation

### 4.1 Prevent Invalid User Input

Validators provide fast feedback before API call:

```typescript
// In component
<script>
    import { SearchQueryValidator, ValidationError } from '$lib/services/validation.service';

    let query = '';
    let validationError = '';

    function onQueryChange(e) {
        query = e.target.value;
        try {
            SearchQueryValidator.validate(query);
            validationError = '';  // Valid
        } catch (error) {
            if (error instanceof ValidationError) {
                validationError = error.message;
            }
        }
    }

            SearchQueryValidator.validate(query);  // Validate before API call
            const results = await searchService.search(query);
            // ...
        } catch (error) {
            // Handle error
        }
    }
</script>

```

### 4.2 Benefits

- ✅ Instant feedback (no network delay)
- ✅ Reduce unnecessary API calls
- ✅ Better UX (error message before submission)

**Option 1: Provide Defaults**

```typescript
function normalizeSearchResponse(data: any): SearchResponse {
    return {
        query: data.query || '',
        results: (data.results || []).map(r => ({
            id: r.id || '',
            title: r.title || 'Untitled',
            abstract: r.abstract || 'No description',
            relevance_score: r.relevance_score ?? 0
        }))
    };
}
```

**Option 2: Strict Validation**

```typescript
function validateSearchResponse(data: any): void {
    if (!data.query || typeof data.query !== 'string') {
        throw new Error('Missing required field: query');
    }
    if (!Array.isArray(data.results)) {
        throw new Error('Missing required field: results');
    }
    for (const result of data.results) {
        if (!result.id || !result.title) {
            throw new Error('Missing required field in result: id or title');
        }
    }
}
```

## 6. Runtime Validation Libraries

### 6.1 When to Use Zod/Yup

**Current**: Manual validation
```typescript
if (!query || query.length > 1000) {
    throw new Error('Invalid query');
}
```

**With Zod**: Structured validation
```typescript
import { z } from 'zod';

const SearchQuerySchema = z.string()
    .min(1, 'Query cannot be empty')
    .max(1000, 'Query too long')
    .transform(q => q.trim());

// Validate
const validQuery = SearchQuerySchema.parse(userInput);  // Throws if invalid

// Or safe version
const result = SearchQuerySchema.safeParse(userInput);
if (!result.success) {
    console.error('Validation failed:', result.error.issues);
}
```

### 6.2 Benefits of Runtime Validation Libraries

| Aspect | Manual | Zod/Yup |
|--------|--------|---------|
| **Clarity** | Error-prone | Clear schema |
| **Reusability** | Scattered | Defined once |
| **Error Messages** | Custom | Automatic |
| **Transformation** | Manual | Built-in |
| **Performance** | Fast | Overhead |

### 6.3 Recommendation

**Current approach is good for:**
- Simple validation (length, type checks)
- Performance-critical code
- Small number of types

**Consider Zod/Yup if:**
- Complex nested validation
- Many validation rules
- Want reusable schemas
