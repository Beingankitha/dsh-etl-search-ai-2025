## 1. Key Performance Metrics

### 1.1 Backend Performance Metrics

| Metric | Threshold | Why | How Measure |
|--------|-----------|-----|-------------|
| **Search API Latency** | <500ms (p95) | User expects quick results | Request start → response end |
| **Chat API Latency** | <2s (streaming starts) | Chat seems responsive | First token arrival time |
| **Embedding Generation** | <100ms (per query) | Bottleneck for search | LLM call duration |
| **Vector Search** | <200ms (1M vectors) | Largest dataset scale | ChromaDB query time |
| **Database Query** | <50ms | Shouldn't be bottleneck | SQLite query time |
| **Error Rate** | <0.1% | Should rarely fail | failed_requests / total |
| **Concurrency** | 5-10 concurrent | Max simultaneous requests | HTTP connections |
| **Memory Usage** | <512MB | On typical deployment | Process RSS |

## 2. Measuring Backend Performance

### 2.1 Backend Metrics Collection

```python
from time import time
import logging

logger = logging.getLogger(__name__)

class PerformanceMetrics:
    """Track performance metrics for analysis."""
    
    def __init__(self):
        self.search_times = []
        self.embedding_times = []
        self.db_query_times = []
        self.error_count = 0
        self.total_requests = 0
    
    def record_search(self, elapsed_ms: float):
        """Record search latency."""
        self.search_times.append(elapsed_ms)
        self.total_requests += 1
        
        # Log if slow
        if elapsed_ms > 500:
            logger.warning(f"Slow search: {elapsed_ms}ms", extra={
                "metric": "search_latency",
                "value": elapsed_ms,
                "percentile": self.percentile(self.search_times, 0.95)
            })
    
    def get_stats(self):
        """Return performance statistics."""
        if not self.search_times:
            return {}
        
        return {
            "search_avg": sum(self.search_times) / len(self.search_times),
            "search_p95": self.percentile(self.search_times, 0.95),
            "search_p99": self.percentile(self.search_times, 0.99),
            "error_rate": self.error_count / max(self.total_requests, 1)
        }
    
    @staticmethod
    def percentile(values, p):
        """Calculate percentile."""
        if not values:
            return 0
        sorted_vals = sorted(values)
        index = int(len(sorted_vals) * p)
        return sorted_vals[min(index, len(sorted_vals) - 1)]

# Global metrics instance
metrics = PerformanceMetrics()
```

```python
router = APIRouter()

        embed_start = time()
        query_embedding = await embedding_service.embed(query)
        embed_time = (time() - embed_start) * 1000

            "request_id": request_id,
            "metric": "embedding_time",
            "value": embed_time
        })
        
        # Vector search
        search_start = time()
        results = await vector_store.search(query_embedding, top_k=10)
        search_time = (time() - search_start) * 1000
        
        # Total time
        total_time = (time() - start_time) * 1000
        
        logger.info(f"Search completed in {total_time:.2f}ms", extra={
            "request_id": request_id,
            "metric": "search_latency",
            "value": total_time,
            "embedding_ms": embed_time,
            "search_ms": search_time,
            "result_count": len(results)
        })
        
        # Record for stats
        metrics.record_search(total_time)
        
        return SearchResponse(results=results)
        
    except Exception as e:
        elapsed = (time() - start_time) * 1000
        logger.error(f"Search failed after {elapsed:.2f}ms: {e}", extra={
            "request_id": request_id,
            "metric": "search_error"
        })
        metrics.error_count += 1
        raise
```

```typescript
interface SearchMetrics {
    queryInputTime: number;      // User types query
    requestStartTime: number;     // API call starts
    firstResultTime: number;      // First result visible
    renderCompleteTime: number;   // All results rendered
    totalTime: number;
}

export class SearchService {
    private metrics: SearchMetrics = {};
    
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Start timing
        this.metrics.requestStartTime = performance.now();
        
        try {
            logger.debug('Search request started', {
                query,
                timestamp: this.metrics.requestStartTime
            });
            
            // Make API call
            const response = await apiClient.searchDatasets(query, limit);
            
            // Time to first result (API response time)
            const responseTime = performance.now() - this.metrics.requestStartTime;
            
            logger.info('Search completed', {
                query,
                responseTime,
                resultCount: response.results.length,
                metric: 'search_latency',
                value: responseTime
            });
            
            // Alert if slow
            if (responseTime > 500) {
                logger.warn('Slow search detected', {
                    query,
                    responseTime,
                    threshold: 500
                });
            }
            
            return response;
            
        } catch (error) {
            const errorTime = performance.now() - this.metrics.requestStartTime;
            logger.error('Search failed', {
                query,
                errorTime,
                error: error.message
            });
            throw error;
        }
    }
}
```

```typescript
// frontend/src/lib/stores/searchStore.ts

export function createSearchStore(searchService: SearchApiService) {
    const store = writable<SearchState>(initialState);
    
    return {
        executeSearch: async (query: string) => {
            const storeUpdateStart = performance.now();
            
            // Optimistic update
            store.update((state) => ({
                ...state,
                query,
                loading: true,
                error: null
            }));
            
            const optimisticTime = performance.now() - storeUpdateStart;
            logger.debug('Store updated (optimistic)', {
                duration: optimisticTime,
                metric: 'store_update_time'
            });
            
            try {
                const response = await searchService.search(query, 10);
                
                const resultUpdateStart = performance.now();
                store.update((state) => ({
                    ...state,
                    results: response.results || [],
                    loading: false,
                    error: null
                }));
                
                const resultUpdateTime = performance.now() - resultUpdateStart;
                logger.debug('Store updated (results)', {
                    duration: resultUpdateTime,
                    resultCount: response.results.length,
                    metric: 'store_update_time'
                });
                
            } catch (error) {
                store.update((state) => ({
                    ...state,
                    error: error.message,
                    loading: false
                }));
            }
        }
    };
}
```

## 4. Performance Thresholds & Alerts

### 4.1 Backend Thresholds

```python
class PerformanceAlerts:
    """Generate alerts when metrics exceed thresholds."""
    
    THRESHOLDS = {
        'search_latency_p95': 500,      # ms
        'embedding_time': 100,           # ms
        'error_rate': 0.001,             # 0.1%
        'memory_usage': 500,             # MB
        'response_time_p99': 1000        # ms
    }
    
    def check_alerts(self, metrics):
        """Check if any metrics exceed thresholds."""
        alerts = []
        
        if metrics.get('search_p95', 0) > self.THRESHOLDS['search_latency_p95']:
            alerts.append({
                'severity': 'warning',
                'message': f"Search latency {metrics['search_p95']:.0f}ms exceeds threshold"
            })
        
        if metrics.get('error_rate', 0) > self.THRESHOLDS['error_rate']:
            alerts.append({
                'severity': 'critical',
                'message': f"Error rate {metrics['error_rate']*100:.2f}% exceeds threshold"
            })
        
        if metrics.get('memory_mb', 0) > self.THRESHOLDS['memory_usage']:
            alerts.append({
                'severity': 'warning',
                'message': f"Memory usage {metrics['memory_mb']:.0f}MB exceeds threshold"
            })
        
        return alerts
```

### 4.2 Frontend Thresholds

```typescript
class PerformanceMonitor {
    private thresholds = {
        searchLatency: 500,        // ms
        storeUpdate: 10,           // ms
        componentRender: 100,      // ms
        networkRequest: 500        // ms
    };
    
    checkAlert(metric: string, value: number): Alert | null {
        const threshold = this.thresholds[metric];
        
        if (value > threshold) {
            return {
                severity: 'warning',
                metric,
                value,
                threshold,
                message: `${metric} exceeded threshold: ${value}ms > ${threshold}ms`
            };
        }
        
        return null;
    }
}
```
