```typescript
export function createSearchStore(searchService: SearchApiService) {
    const store = writable<SearchState>(initialState);
    
    return {
        executeSearch: async (query: string) => {
            store.update(state => ({
                ...state,
                query,
                loading: true,
                error: null
            }));
            
            try {
                const response = await searchService.search(query, 10);
                store.update(state => ({
                    ...state,
                    results: response.results || [],
                    loading: false,
                    error: null
                }));
            } catch (error) {

                    ...state,
                    error: error.message,
                    loading: false,
                    results: []
                }));
                throw error;
            }
        },
        
        clear: () => {
            store.set(initialState);
        },

    };
}
```

### 2.2 Unit Test: Mock Search Service

```typescript
import { describe, it, expect, vi } from 'vitest';
import { createSearchStore } from '$lib/stores/searchStore';
import type { SearchApiService } from '$lib/services/search.service';

describe('searchStore', () => {
    // Mock search service
    const createMockSearchService = (): SearchApiService => ({
        search: vi.fn(),
        getSuggestions: vi.fn()
    });

    it('should execute search and update store with results', async () => {
        const mockService = createMockSearchService();
        const mockResults = {
            results: [
                { dataset: { id: 'ds-1', title: 'Dataset 1' }, score: 0.95 },
                { dataset: { id: 'ds-2', title: 'Dataset 2' }, score: 0.87 }
            ]
        };
        
        // Mock the API response
        mockService.search.mockResolvedValue(mockResults);
        
        const store = createSearchStore(mockService);
        let state;
        store.subscribe(s => { state = s; })();
        
        // Execute search
        await store.executeSearch('biology');
        
        // Verify API was called
        expect(mockService.search).toHaveBeenCalledWith('biology', 10);
        
        // Verify store state updated correctly
        expect(state.query).toBe('biology');
        expect(state.results).toEqual(mockResults.results);
        expect(state.loading).toBe(false);
        expect(state.error).toBeNull();
    });

    it('should handle search errors and update error state', async () => {
        const mockService = createMockSearchService();
        const error = new Error('Network error');
        
        mockService.search.mockRejectedValue(error);
        
        const store = createSearchStore(mockService);
        let state;
        store.subscribe(s => { state = s; })();
        
        // Attempt search
        try {
            await store.executeSearch('test');
        } catch (e) {
            // Error is expected
        }
        
        // Verify error state
        expect(state.error).toBe('Network error');
        expect(state.loading).toBe(false);
        expect(state.results).toEqual([]);
    });

    it('should show loading state while searching', async () => {
        const mockService = createMockSearchService();
        
        // Create a promise that never resolves to observe loading state
        let resolveSearch: Function;
        mockService.search.mockReturnValue(
            new Promise(resolve => { resolveSearch = resolve; })
        );
        
        const store = createSearchStore(mockService);
        let state;
        store.subscribe(s => { state = s; })();
        
        // Start search (don't await)
        const searchPromise = store.executeSearch('test');
        
        // Immediately check loading state
        expect(state.loading).toBe(true);
        
        // Complete search
        resolveSearch({ results: [] });
        await searchPromise;
        
        expect(state.loading).toBe(false);
    });
});
```

```typescript
describe('Error handling in stores', () => {
    it('should differentiate network errors from validation errors', async () => {
        const mockService = createMockSearchService();

        store.subscribe(s => { state = s; })();

            // Expected
        }
        
        // Network errors should be retryable

        expect(state.results).toEqual([]);
    });

    it('should handle validation errors (400)', async () => {

        const store = createSearchStore(mockService);
        let state;
        store.subscribe(s => { state = s; })();
        
        try {
            await store.executeSearch('x');  // Too short
        } catch (e) {
            // Expected
        }
        
        expect(state.error).toContain('Query too short');
        expect(state.loading).toBe(false);
    });

    it('should handle timeout errors', async () => {
        const mockService = createMockSearchService();

        mockService.search.mockRejectedValue(timeoutError);

        let state;

            // Expected
```

## 5. Testing Framework Choice

### 5.1 Backend: pytest (Current)

**Why pytest is ideal for backend**:
- ✅ Async support with `@pytest.mark.asyncio`
- ✅ Excellent fixture system for setup/teardown
- ✅ Powerful mocking with `unittest.mock` and `pytest-mock`
- ✅ Great integration test support
- ✅ Performance test plugins available
- ✅ Industry standard for Python

### 5.2 Frontend: Vitest (Recommended)

**Why Vitest over Jest for SvelteKit 5**:
- ✅ Native ESM support (matches SvelteKit)
- ✅ Faster execution (5-10x faster than Jest)
- ✅ Better TypeScript support out of box
- ✅ Jest-compatible API (easier migration)
- ✅ Better Svelte component testing
- ✅ Built-in coverage reporting

**Example vitest.config.ts** (should you add it):
```typescript
import { defineConfig } from 'vitest/config';
import { sveltekit } from '@sveltejs/kit/vite';

export default defineConfig({
    plugins: [sveltekit()],
    test: {
        globals: true,           // Global test, expect, etc.
        environment: 'jsdom',    // Browser-like environment
        setupFiles: [],          // Global setup if needed
        include: ['src/**/*.test.ts'],
        coverage: {
            provider: 'v8',
            reporter: ['text', 'json', 'html'],
            exclude: ['node_modules/', 'dist/']
        }
    }
});
```

### 5.3 Integration Tests: Playwright (Recommended for E2E)

**When to use Playwright**:
- ✅ Test real user flows (search → view results → read chat)
- ✅ Test browser interactions (clicks, form fills)
- ✅ Test error handling across components
- ✅ Visual regression testing

**Example**: `tests/e2e/search-flow.spec.ts`
