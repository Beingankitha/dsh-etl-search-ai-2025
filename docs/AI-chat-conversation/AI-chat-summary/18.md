interface SearchState {
    query: string;
    results: SearchResult[];
    loading: boolean;
    error: string | null;

const initialState: SearchState = {
    query: '',
    results: [],
    loading: false,
    error: null,
    hasSearched: false
};

export function createSearchStore(searchService: SearchApiService) {
    const store = writable<SearchState>(initialState);

    // Derived stores for computed state
    const hasResults = derived(store, ($store) => $store.results.length > 0);
    const resultCount = derived(store, ($store) => $store.results.length);
    const isLoading = derived(store, ($store) => $store.loading);

    return {
        subscribe: store.subscribe,
        hasResults: { subscribe: hasResults.subscribe },
        resultCount: { subscribe: resultCount.subscribe },
        isLoading: { subscribe: isLoading.subscribe },

        setQuery: (query: string) => {
            store.update((state) => ({ ...state, query }));
        },

        executeSearch: async (query: string) => {
            store.update((state) => ({
                ...state,
                query,
                loading: true,
                error: null,
                hasSearched: true
            }));

            try {
                const response = await searchService.search(query, 10);
                store.update((state) => ({
                    ...state,
                    results: response.results || [],
                    loading: false,
                    error: null
                }));
            } catch (error) {
                store.update((state) => ({
                    ...state,
                    error: errorMessage,
                    loading: false,
                    results: []
                }));
            }
        },

        clear: () => {
            store.set(initialState);
        }
    };
}
```

interface ChatState {
    messages: ChatMessage[];
    messagesSources: Map<number, SearchResult[]>;  // Message index → sources
    loading: boolean;
    error: string | null;
}

export function createChatStore(chatService: ChatApiService) {
    const store = writable<ChatState>(initialState);

    return {
        sendMessage: async (message: string) => {
            try {
                // Validate message
                if (!chatService.validateMessage(message)) {
                    throw new ApiError(400, 'Invalid message', false);
                }

                // Optimistic update: add user message immediately
                const userMessage: ChatMessage = { role: 'user', content: message };
                store.update((state) => ({
                    ...state,
                    messages: [...state.messages, userMessage],
                    loading: true,
                    error: null
                }));

                // Get current messages for context
                let currentMessages: ChatMessage[] = [];
                store.subscribe((state) => {
                    currentMessages = state.messages;
                })();

                // Send to API and wait for response
                const response = await chatService.sendMessage(message, currentMessages);

                // Add assistant response
                if (response.message) {
                    store.update((state) => {
                        const newMessages = [...state.messages, response.message];
                        return {
                            ...state,
                            messages: newMessages,
                            messagesSources: new Map(state.messagesSources),
                            loading: false,
                            error: null
                        };
                    });
                }
            } catch (error) {
                // Error handling
                store.update((state) => ({
                    ...state,
                    error: errorMessage,
                    loading: false
                }));
            }
        }
    };
}
```

## 2. State Synchronization Patterns

### 2.1 Current Pattern: Client-Initiated Updates

Your application uses a **client-initiated, synchronous** pattern:

```
User Action (search/send message)
    ↓
Client validates locally
    ↓
Optimistic update (add to store)
    ↓
Async API call (fire-and-forget or wait)
    ↓
Server processes
    ↓
If success: Keep optimistic update
If failure: Revert to previous state + show error
```

**When This Works Well**:
- Search operations: Results don't change after fetch
- Chat messages: User message is canonical
- Dataset uploads: Once sent, data is known

**When This Fails**:
- Metadata updates: Dataset info changes on server
- Conflict resolution: Two users editing simultaneously
- Session timeouts: User's data becomes stale

### 2.2 Backend State Model

**Backend Stateless Nature**:

The backend (FastAPI + SQLite) is **stateless**:
- No session-specific data
- No live connections (REST-based)
- Database is source of truth
- Each request independent

## 3. Synchronization Strategies

### 3.1 Strategy 1: Eventual Consistency (Current)

**How It Works**:
1. Client updates local state optimistically
2. API call updates backend
3. On next interaction, client fetches fresh state
4. Eventual consistency: states converge over time

**Implementation**:

```typescript
// Search: Client optimistic, refresh on next search
executeSearch: async (query: string) => {
    // 1. Optimistic update
    store.update(state => ({
        ...state,
        query,
        loading: true,
        results: []  // Clear old results
    }));

    try {
        // 2. Fetch from server
        const response = await searchService.search(query, 10);
        
        // 3. Replace with authoritative data
        store.update(state => ({
            ...state,
            results: response.results,
            loading: false
        }));
    } catch (error) {
        store.update(state => ({
            ...state,
            error: error.message,
            loading: false
        }));
    }
}
```

**Eventual Consistency Timeline**:

```
Time 0:  User searches for "biology"
         Frontend: results = [] (optimistic)
         Backend: DB hasn't been searched yet

Time 100ms: Search API call arrives at backend
            Backend searches database
            Returns 50 matching datasets

Time 200ms: Frontend receives response
            Frontend: results = [50 datasets] (consistent with backend)
            Backend: DB returns 50 datasets
            ✅ States are now consistent

User closes browser without searching again...

...Two hours later:

Time T:  Another user adds new "biology" dataset
         Backend: DB now has 51 matching datasets
         Frontend: Locally still shows 50 (inconsistent)
         
Time T + 1: Original user searches again
            Backend: Returns 51 datasets
            Frontend: Updates to 51
            ✅ States re-converge
```

**Pros**:
- ✅ Simple to implement (current approach)
- ✅ No ongoing connection needed
- ✅ Works well for read-mostly data
- ✅ Low server cost

**Cons**:
- ❌ Stale data between updates
- ❌ User might not know data changed
- ❌ No real-time collaboration
- ❌ Conflicts on simultaneous edits

**When to Use**:
- Search results (semi-static)
- Public datasets (infrequent changes)
- User's own chat history (unique writer)

### 3.2 Strategy 2: Polling

**How It Works**:
1. Client periodically fetches fresh state
2. Compares with local state
3. Updates if changed
4. Repeats every N seconds

**Implementation**:

```typescript
// Poll for new datasets every 30 seconds
async function pollForDatasetUpdates(store: any) {
    const POLL_INTERVAL = 30000;  // 30 seconds
    
    const pollTimer = setInterval(async () => {
        try {
            const freshDatasets = await getDatasets();
            store.update(state => {
                // Only update if data actually changed
                if (JSON.stringify(state.datasets) !== JSON.stringify(freshDatasets)) {
                    console.log('Dataset updates detected, refreshing...');
                    return { ...state, datasets: freshDatasets };
                }
                return state;  // No change, don't trigger re-render
            });
        } catch (error) {
            console.error('Poll failed:', error);
            // Silently fail, retry next interval
        }
    }, POLL_INTERVAL);

    // Cleanup
    return () => clearInterval(pollTimer);
}
```

**Usage in Component**:

```svelte
<script>
    import { onMount, onDestroy } from 'svelte';
    
    let datasetStore = createDatasetStore();
    let unsubscribePoll;

    onMount(() => {
        // Start polling when component mounts
        unsubscribePoll = pollForDatasetUpdates(datasetStore);
    });

    onDestroy(() => {
        // Stop polling when component unmounts
        if (unsubscribePoll) unsubscribePoll();
    });
</script>

<div>
    {#each $datasetStore.datasets as dataset}
        <DatasetCard {dataset} />
    {/each}
</div>
```

**Trade-offs**:

| Aspect | Cost |
|--------|------|
| **Network** | High (constant requests) |
| **Latency to update** | 0-30 seconds (depends on interval) |
| **Server load** | Linear (1 request per client per interval) |
| **Battery drain (mobile)** | High (constant polling) |
| **Implementation** | Simple |

**Optimization: Adaptive Polling**:

```typescript
function createAdaptivePoller(store: any) {
    let pollInterval = 30000;  // Start at 30s
    let changeCount = 0;
    
    setInterval(async () => {
        const fresh = await getDatasets();
        store.subscribe(state => {
            if (JSON.stringify(state) !== JSON.stringify(fresh)) {
                changeCount++;
                // Data changed frequently, poll faster
                if (changeCount > 5) {
                    pollInterval = 5000;  // Poll every 5s
                }
            } else {
                changeCount = 0;
                // Data stable, poll slower to save bandwidth
                if (pollInterval < 60000) {
                    pollInterval = 60000;  // Poll every 60s
                }
            }
        })();
        
        // Sleep for pollInterval
        await sleep(pollInterval);
    }, 1000);
}
```

**When to Use Polling**:
- Dashboard data (updates important but not critical)
- Analytics (refresh every minute is acceptable)
- Budget-conscious backend (no WebSocket infrastructure)

### 3.3 Strategy 3: WebSockets (Real-Time)

**How It Works**:
1. Establish bidirectional connection
2. Server pushes updates when data changes
3. Client updates immediately
4. Always-on synchronization

**Architecture**:

```
Client                              Server
  |                                   |
  |-------- Connect WS ------>        |
  |<----- Connection ACK ------       |
  |                                   |
  | (Background sync)                 |
  |                                   |
  | User searches                     |
  |-------- Send message ----->       |
  |<----- Search results ------       | Update state
  |                                   |
  | (Other user modifies data)        |
  |<----- Data changed event --       |
  |                                   |
  | (Always in sync)                  |
```

**Implementation with Svelte**:

```typescript
// WebSocket connection manager
export class RealtimeSync {
    ws: WebSocket | null = null;
    reconnectDelay = 1000;
    maxReconnectDelay = 30000;

    constructor(private apiUrl: string) {}

    async connect(): Promise<void> {
        return new Promise((resolve, reject) => {
            try {
                const wsUrl = this.apiUrl.replace('http', 'ws');
                this.ws = new WebSocket(wsUrl + '/ws');

                this.ws.onopen = () => {
                    console.log('WebSocket connected');
                    this.reconnectDelay = 1000;  // Reset on successful connect
                    resolve();
                };

                this.ws.onerror = (error) => {
                    console.error('WebSocket error:', error);
                    reject(error);
                };

                this.ws.onclose = () => {
                    console.log('WebSocket closed, reconnecting...');
                    this.reconnect();
                };
            } catch (error) {
                reject(error);
            }
        });
    }

    async reconnect(): Promise<void> {
        await new Promise(resolve => setTimeout(resolve, this.reconnectDelay));
        this.reconnectDelay = Math.min(this.reconnectDelay * 2, this.maxReconnectDelay);
        await this.connect();
    }

    onMessage(callback: (data: any) => void): void {
        if (this.ws) {
            this.ws.onmessage = (event) => {
                const data = JSON.parse(event.data);
                callback(data);
            };
        }
    }

    send(type: string, payload: any): void {
        if (this.ws && this.ws.readyState === WebSocket.OPEN) {
            this.ws.send(JSON.stringify({ type, payload }));
        }
    }

    close(): void {
        if (this.ws) {
            this.ws.close();
        }
    }
}

// Integration with store
export function createRealtimeChatStore(chatService: ChatApiService) {
    const store = writable<ChatState>(initialState);
    const sync = new RealtimeSync(import.meta.env.VITE_API_URL);

    // Listen for server-side updates
    sync.onMessage((message) => {
        if (message.type === 'message_added') {
            store.update(state => ({
                ...state,
                messages: [...state.messages, message.payload]
            }));
        }
        if (message.type === 'conversation_updated') {
            store.update(state => ({
                ...state,
                metadata: message.payload
            }));
        }
    });

    return {
        sendMessage: async (text: string) => {
            // Optimistic update
            store.update(state => ({
                ...state,
                messages: [...state.messages, { role: 'user', content: text }]
            }));

            // Send to server (server will broadcast to all connected clients)
            sync.send('message', { text });

            // Wait for server confirmation
            // (Not needed if WebSocket already confirms)
        },

        subscribe: store.subscribe,
        connect: () => sync.connect(),
        disconnect: () => sync.close()
    };
}
```

interface OptimisticMessage {
    id: string;
    content: string;
    pending: boolean;
    error?: string;
}

sendMessage: async (text: string) => {
    const tempId = generateId();
    
    try {
        // Optimistic update with temp ID

            id: tempId,
            content: text,
            pending: true
        };

            ...state,
            messages: state.messages.map(m =>

                    ? { ...m, error: error.message, pending: false }
                    : m
            ),
            loading: false
        }));
    }
}

## 5. Conflict Resolution

### 5.1 Conflict Types

**Write-Write Conflict**:
```
User A updates dataset title: "Biology 101" → "Advanced Biology"
User B updates title: "Biology 101" → "Intro Biology"

Server receives A first: Dataset.title = "Advanced Biology"
Server receives B: Dataset.title = "Intro Biology" (overwrites A)

Result: A's update lost, B's view is correct
```

**Read-Write Conflict**:
```
User A reads: Search results = [50 datasets]
User B adds new dataset
User A sees stale results = [50 datasets]
User B sees fresh results = [51 datasets]
```

### 5.2 Strategies to Prevent Conflicts

**Strategy 1: Timestamps + Last-Write-Wins**

```typescript
interface Dataset {
    id: string;
    title: string;
    updatedAt: number;  // Unix timestamp
}

// On update:
async function updateDataset(dataset: Dataset): Promise<void> {
    const updatedDataset = {
        ...dataset,
        updatedAt: Date.now()
    };
    
    const response = await api.put(`/datasets/${dataset.id}`, updatedDataset);
    
    // Server applies only if client's timestamp is newer than server's
    // Response contains server's version (authoritative)
    store.update(state => ({
        ...state,
        dataset: response  // Accept server's version
    }));
}
```

**Strategy 2: Versioning + Conflict Detection**

```typescript
interface Dataset {
    id: string;
    title: string;
    version: number;
}

async function updateDataset(dataset: Dataset): Promise<void> {
    try {
        const response = await api.put(
            `/datasets/${dataset.id}`,
            { title: dataset.title, version: dataset.version }
        );
        
        // Server returns 200 if version matches
        store.update(state => ({
            ...state,
            dataset: response,
            version: response.version
        }));
    } catch (error) {
        if (error.statusCode === 409) {  // Conflict
            // Get server's latest version
            const latest = await api.get(`/datasets/${dataset.id}`);
            
            // Show merge dialog to user
            showMergeDialog({
                local: dataset,
                server: latest,
                onMerge: (merged) => updateDataset(merged)
            });
        }
    }
}
```

**Strategy 3: Operational Transformation** (Advanced)

Used by Google Docs:
- Track both update and position
- Automatically resolve non-conflicting edits
- Transform operations if they conflict

Example:
```
User A: Insert "bio" at position 10
User B: Insert "data" at position 5

If B's insert comes first, A's position shifts 10+4=14
Result: Both inserts work, no data lost
```

### 5.3 Your Application's Conflict Strategy

**Search results**: No conflict (read-only)
```
Backend state is authoritative
Frontend accepts whatever server returns
No merge needed
```

**Chat messages**: No conflict (append-only)
```
Each user has their own chat history
Messages ordered by creation time
No simultaneous edits possible
```

### 6.2 Background Refresh Strategy

```typescript
export function createAutoRefreshStore(store: any, intervalMs: number = 300000) {
    let refreshTimer: NodeJS.Timeout;

    const startAutoRefresh = () => {
        refreshTimer = setInterval(async () => {
            console.log('Auto-refreshing search results...');
            // Current store state
            let query: string;
            store.subscribe(s => { query = s.query; })();
            
            // Refresh without user interaction
            await store.executeSearch(query);
        }, intervalMs);
    };

    const stopAutoRefresh = () => {
        clearInterval(refreshTimer);
    };

    return {
        startAutoRefresh,
        stopAutoRefresh,
        subscribe: store.subscribe
    };
}

// Usage:
let searchStore = createAutoRefreshStore(
    createSearchStore(searchService),
    600000  // Refresh every 10 minutes
);

onMount(() => {
    searchStore.startAutoRefresh();
});

onDestroy(() => {
    searchStore.stopAutoRefresh();
});
```

## 8. Synchronization Decision Tree

```
┌─ What kind of data?
│
├─ Read-only (search results)
│  └─ Use: Eventual Consistency (current)
│     └─ User must search again for fresh results
│
├─ Append-only (chat history)
│  └─ Use: Eventual Consistency (current)
│     └─ User can see their sent messages immediately
│
├─ Frequently updated (user profile)
│  ├─ Single writer (only user edits own profile)
│  │  └─ Use: Last-Write-Wins with timestamps
│  │
│  └─ Multiple writers (collaboration)
│     └─ Use: Polling or WebSocket
│
└─ Real-time critical (live chat)
   └─ Use: WebSocket + Operational Transformation
      └─ Handles conflicts automatically
```

## 10. Monitoring State Sync Health

```typescript
// Track state synchronization metrics
interface SyncMetrics {
    lastSync: number;           // When was last successful sync?
    staleness: number;          // Milliseconds since last refresh
    conflictCount: number;      // How many conflicts detected?
    optimisticUpdateFailures: number;  // How many reverted?
}

function createMetricsStore() {
    return {
        recordSync: () => {
            metrics.lastSync = Date.now();
        },
        
        recordFailure: () => {
            metrics.optimisticUpdateFailures++;
            if (metrics.optimisticUpdateFailures > 10) {
                console.warn('High failure rate, consider server health check');
            }
        },

        getMetrics: () => ({
            staleness: Date.now() - metrics.lastSync,
            ...metrics
        })
    };
}

// Alert if data too stale
setInterval(() => {
    const metrics = metricsStore.getMetrics();
    if (metrics.staleness > 600000) {  // 10 minutes
        console.warn('Data is very stale, force refresh?');
    }
}, 60000);  // Check every minute
```
