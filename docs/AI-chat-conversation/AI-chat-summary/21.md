## 1. Error Class Hierarchy

### 1.1 Why Hierarchy Matters

**Without hierarchy (Bad)**:
```typescript
if (error.message.includes('timeout')) {
    // Handle timeout
} else if (error.message.includes('network')) {
    // Handle network
} else if (error.message.includes('unauthorized')) {
    // Handle auth
}
// Fragile: depends on error message text
```

**With hierarchy (Good)**:
```typescript
if (error instanceof TimeoutError) {
    // Handle timeout
} else if (error instanceof NetworkError) {
    // Handle network
} else if (error instanceof AuthError) {
    // Handle auth
}
// Robust: depends on error type
```

### 1.2 Your Implementation

Your frontend implements a clean error hierarchy:

**File**: [frontend/src/lib/api/errors.ts](frontend/src/lib/api/errors.ts)

```typescript
/**
 * Base error class for all API errors
 */
export class ApiError extends Error {
    constructor(
        public status: number,           // HTTP status code
        message: string,
        public retryable: boolean = false,  // Can retry?
        public cause?: Error             // Root cause
    ) {
        super(message);
        this.name = 'ApiError';
        Object.setPrototypeOf(this, ApiError.prototype);
    }
}

/**
 * Network-related errors (transient, retryable)
 */
export class NetworkError extends ApiError {
    constructor(message: string, cause?: Error) {
        super(0, message, true, cause);  // retryable=true
        this.name = 'NetworkError';
        Object.setPrototypeOf(this, NetworkError.prototype);
    }
}

/**
 * Request validation errors (client error, not retryable)
 */
export class ValidationError extends ApiError {
    constructor(message: string, public details?: Record<string, string>) {
        super(400, message, false);  // retryable=false
        this.name = 'ValidationError';
        Object.setPrototypeOf(this, ValidationError.prototype);
    }
}

/**
 * Authentication/Authorization errors (not retryable)
 */
export class AuthError extends ApiError {
    constructor(status: number, message: string) {
        super(status, message, false);  // Not retryable
        this.name = 'AuthError';
        Object.setPrototypeOf(this, AuthError.prototype);
    }
}

/**
 * Server errors (5xx, retryable)
 */
export class ServerError extends ApiError {
    constructor(status: number, message: string) {
        super(status, message, status >= 500);  // Retryable only if 5xx
        this.name = 'ServerError';
        Object.setPrototypeOf(this, ServerError.prototype);
    }
}

/**
 * Timeout errors (retryable)
 */
export class TimeoutError extends ApiError {
    constructor(message: string = 'Request timeout') {
        super(408, message, true);  // retryable=true
        this.name = 'TimeoutError';
        Object.setPrototypeOf(this, TimeoutError.prototype);
    }
}

/**
 * Not found errors (not retryable)
 */
export class NotFoundError extends ApiError {
    constructor(message: string = 'Resource not found') {
        super(404, message, false);  // Not retryable
        this.name = 'NotFoundError';
        Object.setPrototypeOf(this, NotFoundError.prototype);
    }
}

/**
 * Check if an error is retryable
 */
export function isRetryable(error: unknown): boolean {
    if (error instanceof ApiError) {
        return error.retryable;
    }
    return false;
}
```

### 1.3 Hierarchy Visualization

```
Error (JavaScript built-in)
  ↓
ApiError (base class for all API errors)
  ├─ NetworkError (status 0, retryable)
  ├─ ValidationError (status 400, not retryable)
  ├─ AuthError (status 401/403, not retryable)
  ├─ ServerError (status 5xx, retryable)
  ├─ TimeoutError (status 408, retryable)
  └─ NotFoundError (status 404, not retryable)
```

### 1.4 Benefits

✅ **Type Safety**: `instanceof` checks work
✅ **Metadata**: Each error has status, retryable flag
✅ **Cause Chain**: `cause` field for debugging
✅ **Extensible**: Easy to add new error types

```python
class RecoverableError(Exception):
    """Base class for errors that can be recovered from"""
    pass

class NetworkError(RecoverableError):
    """Network-related error"""
    pass

class ParsingError(RecoverableError):
    """Metadata parsing error"""
    pass

class DatabaseError(RecoverableError):
    """Database operation error"""
    pass

class NonRecoverableError(Exception):
    """Base class for errors that cannot be recovered from"""
    pass

class ValidationError(NonRecoverableError):
    """Data validation error"""
    pass
```

**Usage**:

```python
def handle_etl_error(error: Exception) -> None:
    if isinstance(error, RecoverableError):
        # Retry with backoff
        retry_handler.execute(operation)
    elif isinstance(error, NonRecoverableError):
        # Fail immediately
        raise error
    else:
        # Unknown error
```

## 4. Error Handling Patterns

### 4.1 Centralized Error Handling (Recommended)

**Create error handler service**:

```typescript
// errorHandler.ts
export class ErrorHandler {
    static handle(error: unknown): ErrorResponse {
        if (error instanceof NetworkError) {
            return {
                type: 'network',
                message: 'Network connection lost. Please check your internet.',
                retryable: true,
                action: 'retry'
            };
        } else if (error instanceof ValidationError) {
            return {
                type: 'validation',
                message: 'Invalid input. Please check your search query.',
                retryable: false,
                action: 'fix_input',
                details: error.details
            };
        } else if (error instanceof AuthError) {
            return {
                type: 'auth',
                message: 'Session expired. Please log in again.',
                retryable: false,
                action: 'login'
            };
        } else if (error instanceof ServerError) {
            return {
                type: 'server',
                message: 'Server error. Please try again later.',
                retryable: true,
                action: 'retry'
            };
        } else if (error instanceof TimeoutError) {
            return {
                type: 'timeout',
                message: 'Request timeout. Please try again.',
                retryable: true,
                action: 'retry'
            };
        } else {
            return {
                type: 'unknown',
                message: 'An unknown error occurred.',
                retryable: false,
                action: 'report'
            };
        }
    }
}

// In component
async function search() {
    try {
        const results = await searchService.search(query);
    } catch (error) {
        const response = ErrorHandler.handle(error);
        showToast(response.message);
        
        if (response.action === 'retry') {
            // Show retry button
        } else if (response.action === 'fix_input') {
            // Focus input field
        }
    }
}
```

## 5. Error Prevention Strategies

### 5.1 Fail Fast on Invalid Input

```typescript
// Validate early
async search(query: string) {
    if (!query || query.trim().length === 0) {
        throw new ValidationError('Query cannot be empty');

    if (query.length > 1000) {

    }

    // Now safe to proceed
    return apiClient.search(query);
```

## 6. Logging Errors Safely

**BAD**:


## 7. Production Error Monitoring

### 7.1 Error Metrics

```typescript
// Track error patterns
class ErrorMetrics {
    private errors: Map<string, ErrorStats> = new Map();

    recordError(error: ApiError) {
        const key = error.constructor.name;
        const stats = this.errors.get(key) || { count: 0, lastSeen: null };
        
        stats.count++;
        stats.lastSeen = new Date();
        
        this.errors.set(key, stats);
        
        // Alert if pattern detected
        if (stats.count > 10) {
            logger.warn(`High error rate for ${key}`);
        }
    }

    getStats() {
        return Object.fromEntries(this.errors);
    }
}
```

### 7.2 Error Reporting

```typescript
// Send errors to monitoring service
async function reportError(error: ApiError) {
    try {
        await fetch('https://monitoring.service.com/errors', {
            method: 'POST',
            body: JSON.stringify({
                type: error.constructor.name,
                message: error.message,
                status: error.status,
                userAgent: navigator.userAgent,
                url: window.location.href,
                timestamp: new Date().toISOString()
            })
        });
    } catch (e) {
        // Don't fail if monitoring is down
        console.error('Failed to report error:', e);
    }
}
```
