### 1.1 Frontend Search Query Validation

**File**: [frontend/src/lib/services/validation.service.ts](frontend/src/lib/services/validation.service.ts#L1-L100)

```typescript
/**
 * Input Validation Utilities
 * 
 * Centralized validation logic for:
 * - Search queries
 * - Chat messages
 * - API responses
 */

export class SearchQueryValidator {
    static readonly MIN_LENGTH = 1;
    static readonly MAX_LENGTH = 1000;

    static validate(query: string): void {
        const trimmed = query.trim();

        if (trimmed.length === 0) {
            throw new ValidationError('query', 'Search query cannot be empty');
        }

        if (trimmed.length < this.MIN_LENGTH) {
            throw new ValidationError('query', `Query must be at least ${this.MIN_LENGTH} character`);
        }

        if (trimmed.length > this.MAX_LENGTH) {
            throw new ValidationError('query', `Query cannot exceed ${this.MAX_LENGTH} characters`);
        }
    }

    static sanitize(query: string): string {
        return query
            .trim()
            .replace(/\s+/g, ' ')  // Collapse whitespace
            .slice(0, this.MAX_LENGTH);
    }
}

export class ChatMessageValidator {
    static readonly MIN_LENGTH = 1;
    static readonly MAX_LENGTH = 5000;

    static validate(message: string): void {
        const trimmed = message.trim();

        if (trimmed.length === 0) {
            throw new ValidationError('message', 'Message cannot be empty');
        }

        if (trimmed.length < this.MIN_LENGTH) {
            throw new ValidationError('message', `Message must be at least ${this.MIN_LENGTH} character`);
        }

        if (trimmed.length > this.MAX_LENGTH) {
            throw new ValidationError('message', `Message cannot exceed ${this.MAX_LENGTH} characters`);
        }
    }

    static sanitize(message: string): string {
        return message
            .trim()
            .slice(0, this.MAX_LENGTH);
    }
}
```

## 3. Authentication & Token Storage

### 3.1 Token Storage Strategy

| Storage | Security | Pros | Cons |
|---------|----------|------|------|
| **localStorage** | ❌ Low | Persists across sessions | Vulnerable to XSS |
| **sessionStorage** | ⚠️ Medium | Clears on tab close | Still XSS-vulnerable |
| **In-Memory** | ✅ High | Not accessible to JS | Lost on page refresh |
| **HttpOnly Cookie** | ✅ High | Can't access from JS | Requires server coordination |

### 3.2 Recommended: Memory + Refresh Token Rotation

```typescript
// frontend/src/lib/auth/token-manager.ts

class TokenManager {
    private accessToken: string | null = null;  // In-memory only
    private readonly REFRESH_INTERVAL = 4 * 60 * 1000;  // 4 minutes
    private refreshTimer: NodeJS.Timeout | null = null;
    
    /**
     * Set tokens from login response.
     * Access token kept in memory (cleared on page refresh).
     * Refresh token can be sent as HttpOnly cookie by backend.
     */
    setAccessToken(token: string): void {
        this.accessToken = token;
        this.startRefreshTimer();
        
        logger.info('Access token set, refresh timer started');
    }
    
    /**
     * Get current access token (null if expired/logged out).
     */
    getAccessToken(): string | null {
        return this.accessToken;
    }
    
    /**
     * Refresh token before expiration.
     */
    private startRefreshTimer(): void {
        if (this.refreshTimer) {
            clearTimeout(this.refreshTimer);
        }
        
        this.refreshTimer = setTimeout(() => {
            this.refreshAccessToken();
        }, this.REFRESH_INTERVAL);
    }
    
    /**
     * Get new access token using refresh token (in HttpOnly cookie).
     */
    private async refreshAccessToken(): Promise<void> {
        try {
            const response = await fetch('/api/auth/refresh', {
                method: 'POST',
                credentials: 'include'  // Send HttpOnly refresh cookie
                // Note: Token in response body (can be in-memory)
            });
            
            const { access_token } = await response.json();
            this.setAccessToken(access_token);
            
            logger.debug('Access token refreshed');
        } catch (error) {
            logger.error('Token refresh failed, logging out', { error: error.message });
            this.logout();
        }
    }
    
    /**
     * Clear tokens (logout).
     */
    logout(): void {
        this.accessToken = null;
        if (this.refreshTimer) {
            clearTimeout(this.refreshTimer);
        }
    }
}

export const tokenManager = new TokenManager();
```

### 3.3 Backend HttpOnly Cookie Setup

**File**: [backend/src/api/app.py](backend/src/api/app.py#L60-L150)

```python
from fastapi import Response
from datetime import datetime, timedelta, timezone

@router.post('/auth/login')
async def login(credentials: LoginRequest) -> LoginResponse:
    """
    Login endpoint.
    Sets HttpOnly refresh token in cookie.
    Returns access token in JSON (for in-memory storage).
    """
    user = authenticate(credentials.username, credentials.password)
    
    if not user:
        raise HTTPException(status_code=401, detail='Invalid credentials')
    
    # Generate tokens
    access_token = create_access_token(user.id, expires_in=5*60)  # 5 min
    refresh_token = create_refresh_token(user.id, expires_in=7*24*60*60)  # 7 days
    
    response = JSONResponse({
        'access_token': access_token,
        'token_type': 'bearer'
    })
    
    # Set refresh token in HttpOnly cookie (secure against XSS)
    response.set_cookie(
        key='refresh_token',
        value=refresh_token,
        max_age=7*24*60*60,  # 7 days
        httponly=True,       # Not accessible from JavaScript
        secure=True,         # HTTPS only
        samesite='strict'    # CSRF protection
    )
    
    logger.info(f"User {user.id} logged in", extra={"request_id": request_id})
    return response


@router.post('/auth/refresh')
async def refresh_token(request: Request) -> dict:
    """
    Refresh access token using HttpOnly refresh token.
    Frontend cannot access the refresh token.
    """
    refresh_token = request.cookies.get('refresh_token')
    
    if not refresh_token:
        raise HTTPException(status_code=401, detail='Refresh token missing')
    
    try:
        payload = verify_refresh_token(refresh_token)
        user_id = payload['user_id']
    except Exception as e:
        logger.error(f"Invalid refresh token: {e}")
        raise HTTPException(status_code=401, detail='Invalid refresh token')
    
    # Issue new access token
    access_token = create_access_token(user_id, expires_in=5*60)
    
    return {
        'access_token': access_token,
        'token_type': 'bearer'
    }
```

### 3.4 HTTP Headers for Security

```python
# Add security headers to all responses
from fastapi import FastAPI

app = FastAPI()

@app.middleware("http")
async def add_security_headers(request: Request, call_next):
    response = await call_next(request)
    
    # Prevent XSS attacks
    response.headers["X-Content-Type-Options"] = "nosniff"
    response.headers["X-Frame-Options"] = "DENY"
    response.headers["X-XSS-Protection"] = "1; mode=block"
    
    # Prevent MIME type sniffing
    response.headers["Content-Security-Policy"] = (
        "default-src 'self'; "
        "script-src 'self' 'unsafe-inline'; "
        "style-src 'self' 'unsafe-inline'"
    )
    
    # Disable caching for sensitive pages
    response.headers["Cache-Control"] = "no-store, no-cache, must-revalidate"
    response.headers["Pragma"] = "no-cache"
    response.headers["Expires"] = "0"
    
    return response
```

### 7.1 Current Security Grade: A-

**Strengths**:
- ✅ Input validation for search queries and chat messages
- ✅ SQL injection prevention via ORM
- ✅ XSS prevention via Svelte auto-escaping
- ✅ CORS configuration with environment-based origins
- ✅ Security header recommendations
- ✅ HttpOnly cookie strategy for tokens
- ✅ Rate limiting via error recovery (retry backoff)

**Gaps**:
- ❌ No Web Application Firewall (WAF)
- ❌ No rate limiting at API level
- ❌ No Content Security Policy (CSP) strict mode
- ❌ No subresource integrity (SRI) for CDN resources
