export interface SearchApiService {
    search(query: string, limit?: number): Promise<SearchResponse>;
    validateQuery(query: string): boolean;
    sanitizeQuery(query: string): string;
    getFromCache(query: string): SearchResponse | null;
    setCache(query: string, response: SearchResponse): void;
}

export class DefaultSearchApiService implements SearchApiService {
    // Owns: Query sanitization
    sanitizeQuery(query: string): string {
        return query.trim().toLowerCase().slice(0, 200);
    }

    // Owns: Query validation
    validateQuery(query: string): boolean {
        const trimmed = query.trim();
        return trimmed.length > 0 && trimmed.length <= 200;
    }

    // Owns: Caching
    private cache = new Map<string, { data: SearchResponse; timestamp: number }>();
    
    getFromCache(query: string): SearchResponse | null {
        const cached = this.cache.get(query);
        if (cached && Date.now() - cached.timestamp < 5 * 60 * 1000) {
            return cached.data;
        }
        return null;
    }

    setCache(query: string, response: SearchResponse): void {
        this.cache.set(query, { data: response, timestamp: Date.now() });
    }

    // Owns: Search orchestration
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        const sanitizedQuery = this.sanitizeQuery(query);
        if (!sanitizedQuery) {
            throw new ApiError(400, 'Search query cannot be empty', false);
        }

        const cached = this.getFromCache(sanitizedQuery);
        if (cached) {
            return cached;
        }

        try {
            const response = await apiClient.searchDatasets(sanitizedQuery, limit);
            this.setCache(sanitizedQuery, response);
            return response;
        } catch (error) {
            throw new ApiError(500, `Search failed: ${error}`, false);
        }
    }
}
```

✅ **What it owns**: Query validation, sanitization, caching, API orchestration

class SearchService:
    """Semantic search over vector store."""
    
    def __init__(
        self,
        embedding_service: EmbeddingService,
        vector_store: VectorStore,
        unit_of_work: UnitOfWork,
    ):
        self.embedding_service = embedding_service
        self.vector_store = vector_store
        self.unit_of_work = unit_of_work

    @with_span("search")
    def search(
        self,
        query: str,
        top_k: int = 10,
        collection: str = "datasets"
    ) -> List[SearchResult]:
        """Semantic search with vector similarity."""
        
        # 1. Query embedding (delegates to EmbeddingService)
        embedding = self.embedding_service.embed(query)
        
        # 2. Vector similarity search
        search_results = self.vector_store.search(
            embedding,
            top_k=top_k,
            collection=collection
        )
        
        # 3. Enrich results with dataset metadata
        results_with_metadata = []
        for result in search_results:
            dataset = self.unit_of_work.dataset_repo.get(result.dataset_id)
            results_with_metadata.append(
                SearchResult(dataset=dataset, score=result.score)
            )
        
        return results_with_metadata
```

✅ **What it owns**: Query orchestration, embedding coordination, metadata enrichment

## 2. Service Responsibilities Matrix

### 2.1 Should This Service Own It?

| Responsibility | Frontend SearchService | Backend SearchService | Backend ChatService | Verdict |
|----------------|------------------------|----------------------|----------------------|---------|
| **Query validation** | ✅ | ✅ | ✅ (msg validation) | Yes - entry point |
| **Query sanitization** | ✅ | ❌ | ❌ | Depends on layer |
| **Caching** | ✅ | ❌ | ❌ | Frontend (client-side) |
| **API client calls** | ✅ | ❌ | ❌ | API orchestration |
| **Data transformation** | ❌ | ✅ | ✅ | Service layer |
| **Business logic orchestration** | ✅ | ✅ | ✅ | Definitely yes |
| **Persistence** | ❌ | ❌ | ❌ | Repository layer |
| **Logging/Tracing** | ✅ | ✅ | ✅ | Cross-cutting |
| **Dependency injection** | ✅ | ✅ | ✅ | Constructor |
| **Error handling** | ✅ | ✅ | ✅ | Service-specific |

### 2.2 What Should NOT Be in Services

| Anti-Pattern | Why Not | Where Instead |
|--------------|---------|----------------|
| **SQL queries** | Couples to database | Repository layer |
| **Component rendering** | Breaks separation | Component layer |
| **HTTP routing** | Breaks layering | API routes layer |
| **Authentication tokens** | Security risk | Auth middleware |
| **UI state management** | Separate concern | Store layer |
| **Formatting for display** | Presentation logic | Component/Model |
| **Direct database access** | Breaks abstraction | Repository |

## 4. Service Class vs. Functions

### 4.1 Class-Based (What You're Using)

**Advantages**:
- ✅ Encapsulation (private state, public methods)
- ✅ Dependency injection (pass dependencies in constructor)
- ✅ Testability (easy to mock)
- ✅ Inheritance (services can extend base classes)
- ✅ Interface contracts (services implement interfaces)

**Disadvantages**:
- ⚠️ More boilerplate
- ⚠️ Memory overhead (object instances)

**When to use**: **Most service classes** should be class-based for dependency injection

### 4.2 Function-Based

**Advantages**:
- ✅ Simpler syntax
- ✅ Less boilerplate
- ✅ Easier for pure functions

**Disadvantages**:
- ❌ Harder to manage dependencies
- ❌ Global state problems
- ❌ Harder to test

**When to use**: Only for **pure utility functions** (not services)

### 4.3 Hybrid (Recommended)

Your approach is correct:

## 7. Caching Strategy

### 7.1 Frontend Caching (SearchService)

**File**: [frontend/src/lib/services/search.service.ts](frontend/src/lib/services/search.service.ts)

```typescript
export class DefaultSearchApiService {
    private cache = new Map<string, { data: SearchResponse; timestamp: number }>();
    private readonly CACHE_TTL = 5 * 60 * 1000; // 5 minutes

    getFromCache(query: string): SearchResponse | null {
        const cached = this.cache.get(query);
        if (cached && Date.now() - cached.timestamp < this.CACHE_TTL) {
            return cached.data;
        }
        this.cache.delete(query); // Expire stale entry
        return null;
    }

    setCache(query: string, response: SearchResponse): void {
        this.cache.set(query, { data: response, timestamp: Date.now() });
    }
}
```

**Recommendation**: Add cache clearing:

```typescript
export interface SearchApiService {
    clearCache(): void;
    getCacheSize(): number;
}

export class DefaultSearchApiService implements SearchApiService {
    clearCache(): void {
        this.cache.clear();
    }

        return this.cache.size;
    }
}
```
