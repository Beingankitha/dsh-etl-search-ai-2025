### 2.4 Store SRP

**File**: [frontend/src/lib/stores/searchStore.ts](frontend/src/lib/stores/searchStore.ts)

```typescript
export function createSearchStore(searchService: SearchApiService) {
    const store = writable<SearchState>(initialState);
    
    return {
        // Responsibility: Search state management
        search: async (query: string) => {
            store.update(state => ({
                ...state,
                query,
                loading: true,
                error: null
            }));
            
            try {
                const results = await searchService.search(query);
                store.update(state => ({
                    ...state,
                    results,
                    loading: false
                }));

                store.update(state => ({
                    ...state,
                    error: String(error),
                    loading: false
                }));
            }
        }
    };
}
```

**One reason to change**: If search state management changes (new fields, flow)

✅ **Does NOT contain**: Business logic, API calls, data formatting

## 3. Open/Closed Principle (OCP)

**Extension**: Create new implementations without modifying interface

### 2.3 How to Extend (Without Modifying)

**To support new search type (e.g., hybrid search)**:

```typescript
// ✅ OCP: Extend with new implementation
export class HybridSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Combine semantic + keyword search
        const semanticResults = await this.semanticSearch(query);
        const keywordResults = await this.keywordSearch(query);
        return this.mergeResults(semanticResults, keywordResults);
    }
    
    private async semanticSearch(query: string): Promise<SearchResponse> { ... }
    private async keywordSearch(query: string): Promise<SearchResponse> { ... }
    private mergeResults(...): SearchResponse { ... }
}

// ✅ OCP: Container switches implementations
export class Container {
    private searchService: SearchApiService;
    
    constructor(hybrid: boolean = false) {
        this.searchService = hybrid
            ? new HybridSearchApiService()
            : new DefaultSearchApiService();
    }
}
```

**✅ Closed for modification**: SearchApiService interface unchanged
**✅ Open for extension**: New HybridSearchApiService created without touching existing code

## 4. Liskov Substitution Principle (LSP)

### 4.1 Definition
Subtypes must be **substitutable** for their base types.

### 4.2 Your LSP Implementation

**✅ Good LSP: Service Substitutability**

```typescript
export interface SearchApiService {
    search(query: string, limit?: number): Promise<SearchResponse>;
}

export class DefaultSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Real API implementation
    }
}

export class MockSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Returns mock data
    }
}

// Both are substitutable
let service: SearchApiService;

// Production
service = new DefaultSearchApiService();
const results1 = await service.search("test");

// Testing - same contract, different implementation
service = new MockSearchApiService();
const results2 = await service.search("test");

// Both work the same way from consumer's perspective
```

**Contract respected**: Both return Promise<SearchResponse> with same signature

### 4.3 Bad LSP (What NOT to do)

```typescript
// ❌ BAD: Violates LSP
export class CachedSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Changes contract: sometimes returns cached, sometimes not
        // No error on cache miss, just returns stale data
        // Consumer can't rely on getting fresh data
    }
}

// ❌ BAD: Changes return type
export class AdvancedSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<AdvancedSearchResponse> {
        // Different return type!
        // Not substitutable
    }
}

// ❌ BAD: Different exceptions
export class StrictSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Throws different exceptions than other implementations
        // Contract broken
    }
}
```

### 4.4 Repository LSP

**File**: [backend/src/repositories/base_repository.py](backend/src/repositories/base_repository.py)

```python
class BaseRepository(Generic[T], ABC):
    """All repositories are substitutable."""
    
    def get(self, id: int) -> Optional[T]:
        # Contract: Returns entity or None
        
    def find_all(self) -> List[T]:
        # Contract: Returns list (possibly empty)
        
    def insert(self, entity: T) -> Optional[int]:
        # Contract: Returns ID or None

# Specific repositories maintain same contract
class DatasetRepository(BaseRepository[Dataset]):
    def get(self, id: int) -> Optional[Dataset]:
        # Returns Dataset or None - same contract
        
class MetadataDocumentRepository(BaseRepository[MetadataDocument]):
    def get(self, id: int) -> Optional[MetadataDocument]:
        # Returns MetadataDocument or None - same contract
```

**✅ LSP**: Both repositories are interchangeable in code expecting BaseRepository

## 5. Interface Segregation Principle (ISP)

### 5.1 Definition
Clients should depend on **focused interfaces**, not monolithic ones.

### 5.2 Your ISP Implementation

**✅ Good ISP: Focused Service Interfaces**

```typescript
// Instead of monolithic SearchService
export interface SearchApiService {
    search(query: string, limit?: number): Promise<SearchResponse>;
    validateQuery(query: string): boolean;
    sanitizeQuery(query: string): string;
    getFromCache(query: string): SearchResponse | null;
    setCache(query: string, response: SearchResponse): void;
}

// Split into focused roles
export interface Searchable {
    search(query: string, limit?: number): Promise<SearchResponse>;
}

export interface QueryValidator {
    validateQuery(query: string): boolean;
    sanitizeQuery(query: string): string;
}

export interface Cacheable {
    getFromCache(query: string): SearchResponse | null;
    setCache(query: string, response: SearchResponse): void;
}

// Component depends only on what it needs
export class SearchComponent {
    constructor(private searchable: Searchable) {}
    
    search(query: string) {
        return this.searchable.search(query);
    }
}

export class SearchFormValidator {
    constructor(private validator: QueryValidator) {}
    
    validate(query: string) {
        return this.validator.validateQuery(query);
    }
}
```

**✅ ISP**: Components don't depend on caching if they don't use it

### 5.3 Your Current Approach (Also Good)

**File**: [frontend/src/lib/services/search.service.ts](frontend/src/lib/services/search.service.ts)

```typescript
export interface SearchApiService {
    search(query: string, limit?: number): Promise<SearchResponse>;
    validateQuery(query: string): boolean;
    sanitizeQuery(query: string): string;
    getFromCache(query: string): SearchResponse | null;
    setCache(query: string, response: SearchResponse): void;
}
```

✅ **Practical ISP**: Interface is reasonably focused (5 related methods)

**Decision**: Focused interfaces (ISP) vs. practical interfaces (your approach)
- ISP is purer, better for large systems
- Your approach is pragmatic, works fine for 5-method interface

## 6. Dependency Inversion Principle (DIP)

### 6.1 Definition
High-level code should depend on **abstractions**, not low-level implementations.

### 6.2 Your DIP Implementation

**✅ Good DIP: Frontend Container**

**File**: [frontend/src/lib/container.ts](frontend/src/lib/container.ts)

```typescript
// Abstraction: Interface
export interface SearchApiService {
    search(query: string, limit?: number): Promise<SearchResponse>;
}

// Low-level: Implementation
export class DefaultSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // Real implementation
    }
}

// High-level: Container creates implementation
export class Container {
    private searchService: SearchApiService;
    
    constructor() {
        // Depends on abstraction
        this.searchService = new DefaultSearchApiService();
    }
    
    getSearchService(): SearchApiService {
        // Returns abstraction, not implementation
        return this.searchService;
    }
}

// Components depend on abstraction
export function createSearchStore(searchService: SearchApiService) {
    return {
        search: async (query: string) => {
            const results = await searchService.search(query);
            // No knowledge of DefaultSearchApiService
        }
    };
}
```

**✅ DIP**: createSearchStore depends on SearchApiService abstraction, not DefaultSearchApiService

### 6.3 Bad DIP (What NOT to do)

```typescript
// ❌ BAD: High-level depends on low-level implementation
export function createSearchStore() {
    const service = new DefaultSearchApiService();
    // Tightly coupled to DefaultSearchApiService
    // Can't swap implementations
    return {
        search: async (query: string) => {
            const results = await service.search(query);
        }
    };
}

// ❌ BAD: Components depend on concrete types
export class SearchComponent {
    constructor(private service: DefaultSearchApiService) {}
    // Can't swap implementations
}
```

### 6.4 Backend DIP

# Dependency Injection via FastAPI Depends
def get_search_service() -> SearchService:

    embedding_service = EmbeddingService()

@router.post("", response_model=SearchResponse)

## 7. SOLID Principles Integration

### 7.1 How They Work Together

```
DIP (Depend on abstractions)
    ↓
Define SearchApiService interface
    ↓
SRP (Each class has one reason to change)
    ↓
SearchApiService handles search orchestration
    ↓
OCP (Open for extension)
    ↓
Create HybridSearchApiService without modifying SearchApiService
    ↓
LSP (Substitutable types)
    ↓
Use HybridSearchApiService wherever SearchApiService expected
    ↓
ISP (Focused interfaces)
    ↓
Only methods related to search in SearchApiService
```

## 9. Extensibility Example

### 9.1 Adding Advanced Search (OCP + DIP)

```typescript
// ✅ OCP: Extend without modifying existing code
export class AdvancedSearchApiService implements SearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        // New advanced search algorithm
        const parsed = this.parseQuery(query);
        const semantic = await this.semanticSearch(parsed);
        const keyword = await this.keywordSearch(parsed);
        return this.mergeResults(semantic, keyword);
    }
    
    private parseQuery(query: string): ParsedQuery { ... }
    private async semanticSearch(parsed: ParsedQuery): Promise<SearchResponse> { ... }
    private async keywordSearch(parsed: ParsedQuery): Promise<SearchResponse> { ... }
    private mergeResults(semantic: SearchResponse, keyword: SearchResponse): SearchResponse { ... }
}

// ✅ DIP: Container chooses implementation
export class Container {
    getSearchService(advanced: boolean = false): SearchApiService {
        if (advanced) {
            return new AdvancedSearchApiService();
        }
        return new DefaultSearchApiService();
    }
}

// ✅ LSP: Components work with either
export function useSearch(advanced: boolean = false) {
    const service = container.getSearchService(advanced);
    const store = createSearchStore(service);
    return { store };
}
```

## 10. SOLID Implementation Checklist

- [ ] Each class has one reason to change (SRP)
- [ ] Can extend without modifying existing (OCP)
- [ ] Implementations substitutable (LSP)
- [ ] Interfaces focused and minimal (ISP)
- [ ] Dependencies injected (DIP)
- [ ] Services separated by concern
- [ ] Interfaces define contracts
- [ ] Implementations are replaceable
- [ ] No circular dependencies
- [ ] Components depend on abstractions
- [ ] Tests can use mock implementations
- [ ] New features don't require modifying existing
