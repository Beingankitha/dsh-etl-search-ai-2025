### 3.3 Error Handling Helper

**Recommendation**: Extract common error patterns

```typescript
// frontend/src/lib/api/error-handler.ts

export interface ErrorHandlerOptions {
    retryable?: boolean;
    fallback?: any;
    logError?: boolean;
}

export async function handleApiCall<T>(
    operation: () => Promise<T>,
    options: ErrorHandlerOptions = {}
): Promise<T> {
    const { retryable = true, fallback = null, logError = true } = options;
    
    try {
        return await operation();
    } catch (error) {
        if (logError) {
            logger.error('API call failed', { error, retryable });
        }
        
        if (error instanceof ApiError) {
            throw error; // Already handled
        }
        
        // Convert to ApiError
        const apiError = new ApiError(
            500,
            error instanceof Error ? error.message : 'Unknown error',
            retryable
        );
        
        if (fallback !== null) {
            return fallback;
        }
        
        throw apiError;
    }
}

// Usage:
export class DefaultSearchApiService {
    async search(query: string, limit: number = 10): Promise<SearchResponse> {
        return handleApiCall(
            () => apiClient.searchDatasets(query, limit),
            { retryable: true, logError: true }
        );
    }
}
```

def handle_service_errors(service_name: str):
    """Decorator for consistent error handling."""
    def decorator(func):
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            try:
                return await func(*args, **kwargs) if asyncio.iscoroutinefunction(func) else func(*args, **kwargs)
            except Exception as e:
                logger.error(f"{service_name} error: {e}", exc_info=True)
                raise
        return wrapper
    return decorator
