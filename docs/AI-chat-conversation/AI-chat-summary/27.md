## 1. Versioning Strategies

### 1.1 Strategy Comparison

| Strategy | Approach | Pros | Cons | When to Use |
|----------|----------|------|------|------------|
| **URL Path** | `/api/v1/search` | Clear, explicit | Multiple copies of code | Most projects |
| **Query Param** | `/api/search?api-version=1` | Single endpoint | Confusing | Rare |
| **Header** | `Accept: application/vnd.api+json;version=1` | Clean URLs | Not discoverable | APIs only |
| **Media Type** | `/api/search` + `Accept: application/vnd.api.v1+json` | RESTful | Complex | REST purists |

### 1.2 Recommended: URL Path Versioning

**Why for your project**:
- ✅ Explicit and discoverable (`/api/v1/search` vs `/api/v2/search`)
- ✅ Easy to document and test
- ✅ Clear backward compatibility path
- ✅ Simple frontend routing

## 3. Backward Compatibility Strategy

### 3.1 Backward Compatible Changes (Don't need version bump)

✅ **Adding optional fields**:
```python
# Old response
{ "query": "test", "results": [...] }

# New response (backward compatible)
{ 
    "query": "test", 
    "results": [...],
    "metadata": { ... }  # New optional field
}

# Old clients ignore metadata, still work
```

✅ **Adding new endpoints**:
```python
# Add new endpoint without affecting existing ones
@router.get("/search/suggestions")  # New
async def get_suggestions(query: str):
    pass
```

✅ **Deprecating with warning header**:
```python
@router.post("/search")  # Still works in v1
async def search(query: str):
    response = SearchResponse(...)
    response.headers["Deprecation"] = "true"
    response.headers["Sunset"] = "Sun, 01 Jul 2025 00:00:00 GMT"
    response.headers["Link"] = "</api/v2/search>; rel=\"successor-version\""
    return response
```

### 3.2 Breaking Changes (Require version bump)

❌ **Removing fields**:
```python
# v1: Returns title, abstract, keywords
# v2: Returns only title, abstract
# Clients using keywords break
```

❌ **Changing field types**:
```python
# v1: score: number (0.95)
# v2: score: string ("95%")
# Breaks client type checking
```

❌ **Changing response structure**:
```python
# v1: { results: [{ dataset: {...}, score: 0.95 }] }
# v2: { items: [{ data: {...}, confidence: 0.95 }] }
# All clients break
```

```typescript
export class ApiClient {
    private baseUrl: string;
    private version: string;
    
    constructor(baseUrl: string = 'http://localhost:8000', version: string = 'v1') {
        this.baseUrl = baseUrl;
        this.version = version;
    }
    
    // v1 endpoints
    async searchDatasets(query: string, limit: number = 10): Promise<SearchResponse> {
        return httpClient.request<SearchResponse>(
            `${this.baseUrl}/api/${this.version}/search`,
            {
                method: 'POST',
                body: { query, top_k: limit },
                config: { timeout: 30000, retries: 2 }
            }
        );

    // v2 endpoints (if available)
    async searchDatasetsV2(
        query: string,
        filters: SearchFilters = {}

        // New endpoint with different contract

            {
                method: 'POST',
                body: { query, filters },
                config: { timeout: 30000, retries: 2 }
            }
        );
    }
}
```

### 4.2 Version Detection & Migration

```typescript
// Detect available API versions
async function detectApiVersion(): Promise<string> {
    try {
        // Try v2
        const response = await fetch('/api/v2/health');
        if (response.ok) {
            return 'v2';
        }
    } catch (e) {
        // Fall back to v1
    }
    
    try {
        const response = await fetch('/api/v1/health');
        if (response.ok) {
            return 'v1';
        }
    } catch (e) {
        // No API available
    }
    
    throw new Error('API server not available');
}

// On app startup
let apiVersion = await detectApiVersion();
const apiClient = new ApiClient(undefined, apiVersion);
```

### 4.3 Gradual Migration Path

```typescript
// Phase 1: Support v1 and v2 simultaneously
export interface ApiClientConfig {
    preferredVersion: 'v1' | 'v2';
    fallbackVersion: 'v1' | 'v2';
    migrateV1toV2: boolean;  // Auto-convert responses
}

class ApiClient {
    async search(query: string): Promise<SearchResponse> {
        try {
            if (this.config.preferredVersion === 'v2') {
                const response = await this.searchV2(query);
                if (this.config.migrateV1toV2) {
                    return this.convertV2toV1(response);  // Normalize response
                }
                return response;
            }
        } catch (error) {
            if (this.config.fallbackVersion === 'v1') {
                logger.warn('V2 search failed, falling back to V1', { error });
                return this.searchV1(query);
            }
        }
    }
    
    private convertV2toV1(response: SearchResponseV2): SearchResponse {
        // Convert v2 format to v1 for backward compatibility
        return {
            query: response.searchQuery,
            results: response.items.map(item => ({
                dataset: item.data,
                score: item.confidence
            }))
        };
    }
}
```

### 5.2 Deprecation Notices

**Backend**: Deprecation headers in HTTP responses

```python
@router.post("/search")
async def search_v1(query: str):
    """v1 endpoint - deprecated, migrate to v2."""
    response = SearchResponse(results=[])
    
    # Tell clients to migrate
    response.headers["Deprecation"] = "true"
    response.headers["Sunset"] = "Sun, 01 Dec 2025 00:00:00 GMT"
    response.headers["Link"] = '</api/v2/search>; rel="successor-version"'
    response.headers["Warning"] = '299 - "API version 1 is deprecated"'
    
    logger.info("v1 /search used - track deprecation", extra={
        "metric": "deprecated_endpoint_used",
        "endpoint": "/api/v1/search"
    })
    
    return response
```

**Frontend**: Show migration guide

```typescript
// logger.ts
export const logger = Logger.getInstance();

// Detect deprecated endpoints
fetch('/api/v1/search', ...).then(response => {
    if (response.headers.get('Deprecation') === 'true') {
        const sunsetDate = response.headers.get('Sunset');
        logger.warn('API endpoint is deprecated', {
            endpoint: '/api/v1/search',
            sunsetsAt: sunsetDate,
            migrateUrl: '/docs/migrate-to-v2'
        });
    }
});
```

```typescript
// Feature flags by API version
const featureFlags = {
    v1: {
        advancedSearch: false,
        chatWithContext: false,
        datasetMetadata: false

export function isFeatureAvailable(feature: string): boolean {
    const version = apiClient.version;
    return featureFlags[version]?.[feature] ?? false;
```

```python
        """Test v1 search endpoint."""

    def test_search_v2(self, client):
        """Test v2 search endpoint."""
        response = client.post("/api/v2/search", json={"query": "test"})
        assert response.status_code == 200
        # v2 might have different response format
        assert "items" in response.json()
    
    def test_v1_backward_compatibility(self, client):
        """Test v1 response format for backward compatibility."""
        response = client.post("/api/v1/search", json={"query": "test"})
        data = response.json()
        
        # v1 expected fields
        assert "query" in data
        assert "results" in data
        assert all("dataset" in r for r in data["results"])
        assert all("score" in r for r in data["results"])

        """Test that v1 includes deprecation headers."""
```
